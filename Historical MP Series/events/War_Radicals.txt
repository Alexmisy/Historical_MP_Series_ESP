namespace = war_radicals

#event that checks dead from war relative to population and assigns a multiplier based on this amount

war_radicals.1 = {
	type = country_event
	hidden = yes
	
	trigger = {
		any_scope_war = {
			num_country_casualties = {
				target = ROOT
				value > 0
			}
			any_scope_front = { exists = yes }
		}
	}
	
	#figures out how many casualties have occurred, sets multiplier variable
	
	immediate = {
		any_scope_war = {
			num_country_casualties = {
				target = ROOT
				value > one_percent_pop
			}
			any_scope_front = { exists = yes }
		}
		set_variable = {
			name = war_deaths_radical_mult
			value = 1
		}
	}
}
